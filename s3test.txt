var AwsS3 = require ('aws-sdk/clients/s3');
const s3 = new AwsS3 ({
  accessKeyId: '',
  secretAccessKey: '',
  region: 'ap-south-1',
});

const listDirectories = params => {
  return new Promise ((resolve, reject) => {
    const s3params = {
      Bucket: 'bucket-name',
      MaxKeys: 20,
      Delimiter: '/',
    };
    s3.listObjectsV2 (s3params, (err, data) => {
      if (err) {
        reject (err);
      }
      resolve (data);
    });
  });
};
